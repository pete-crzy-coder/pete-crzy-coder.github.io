    dragElement(document.getElementById("main"));
    
    document.addEventListener('keydown', hide)
    
    function dragElement(elmnt) {
        let pos1 = 0, pos2 = 0, pos3 = 0, pos4 = 0;
        if (document.getElementById(elmnt.id + "header")) {
            document.getElementById(elmnt.id + "header").onmousedown = dragMouseDown;
        } else {
            elmnt.onmousedown = dragMouseDown;
        }
    
        function dragMouseDown(e) {
    
            e.preventDefault();
            pos3 = e.clientX;
            pos4 = e.clientY;
            document.onmouseup = closeDragElement;
            document.onmousemove = elementDrag;
        }
    
        function elementDrag(e) {
    
            e.preventDefault();
            pos1 = pos3 - e.clientX;
            pos2 = pos4 - e.clientY;
            pos3 = e.clientX;
            pos4 = e.clientY;
            elmnt.style.top = (elmnt.offsetTop - pos2) + "px";
            elmnt.style.left = (elmnt.offsetLeft - pos1) + "px";
        }
    
        function closeDragElement() {
            document.onmouseup = null;
            document.onmousemove = null;
        }
    }
    function hide(event) {
        if (event.key == 'e' && event.ctrlKey && document.getElementById('main').className != 'hide') {
            document.getElementById('main').className = 'hide'
        } else if (event.key == 'e' && event.ctrlKey) {
            document.getElementById('main').className = ''
        }
    }
    function char() {
        const sel = document.getElementById('sel');
        const i = document.getElementById('i');
        i.src = `https://scratch.mit.edu/projects/${sel.value}/embed`
    }
    document.getElementById('sel').addEventListener('change', char)
    document.getElementById('plus').addEventListener('click', () => {
        const sel = document.getElementById('sel');
        const i = document.getElementById('i');
        sel.innerHTML += `<option value="${prompt('Enter the project id')}">${prompt('Enter the label for the project')}</option>`
